           import com.atlassian.jira.application.ApplicationKeys

 Users.create('jbloggs', 'joe@example.com', 'Joe Bloggs') {
     password = 'supersecretpassword'
     directoryId = 10001
     withApplicationAccess(ApplicationKeys.SERVICE_DESK)
     // etc
 }
